# php

```ini
; 拓展动态库路径
extension_dir = "ext"
; 时区
date.timezone = Asia/Shanghai
; 关掉php的版本显示
expose_php = Off
; 启用外部拓展
enable_dl = On
; fastcgi的自动跳转
cgi.force_redirect = 0
fastcgi.impersonate = 1
; 请求体大小
post_max_size = 256M
; 文件上传
file_uploads = On
; 文件上传大小限制
upload_max_filesize = 256M
```
